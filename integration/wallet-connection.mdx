## Overview

AridGuard uses **Coinbase OnchainKit** to provide a seamless wallet connection experience. OnchainKit offers:

- Smart Wallet support (no seed phrases)
- Traditional wallet compatibility (MetaMask, WalletConnect)
- Built-in identity components (avatars, names, addresses)
- Optimized for Base blockchain

<Note>
  OnchainKit is the recommended way to integrate with Base applications. It provides the best user experience and lowest friction for onboarding.
</Note>

## Installation

Install the required packages:

```bash
npm install @coinbase/onchainkit wagmi viem @tanstack/react-query
```

**Package purposes:**

- `@coinbase/onchainkit`: Wallet UI components and utilities
- `wagmi`: React hooks for Ethereum interactions
- `viem`: TypeScript Ethereum library
- `@tanstack/react-query`: State management for async data

## Provider Setup

Wrap your application with the necessary providers:

```typescript
// app/providers.tsx
"use client";

import { ReactNode } from 'react';
import { OnchainKitProvider } from '@coinbase/onchainkit/OnchainKitProvider';
import { QueryClient, QueryClientProvider } from '@tanstack/react-query';
import { WagmiProvider } from 'wagmi';
import { baseSepolia } from 'wagmi/chains';
import { http, createConfig } from 'wagmi';

// Create wagmi config
const wagmiConfig = createConfig({
  chains: [baseSepolia],
  transports: {
    [baseSepolia.id]: http(),
  },
});

// Create query client
const queryClient = new QueryClient();

interface ProvidersProps {
  children: ReactNode;
}

export function Providers({ children }: ProvidersProps) {
  return (
    <WagmiProvider config={wagmiConfig}>
      <QueryClientProvider client={queryClient}>
        <OnchainKitProvider
          apiKey={process.env.NEXT_PUBLIC_ONCHAINKIT_API_KEY}
          chain={baseSepolia}
        >
          {children}
        </OnchainKitProvider>
      </QueryClientProvider>
    </WagmiProvider>
  );
}
```

**Environment Variables:**

```bash
# .env.local
NEXT_PUBLIC_ONCHAINKIT_API_KEY=your_onchainkit_api_key
```

Get your API key from [Coinbase Developer Platform](https://portal.cdp.coinbase.com/).

## Basic Wallet Connection

Implement a simple connect button:

```typescript
"use client";

import { ConnectWallet, Wallet } from '@coinbase/onchainkit/wallet';
import { useAccount } from 'wagmi';

export default function WalletButton() {
  const { isConnected, address } = useAccount();

  return (
    <div>
      {isConnected ? (
        <p>Connected: {address}</p>
      ) : (
        <Wallet>

}
```

## Checking Connection Status

Use Wagmi hooks to check wallet state:

```typescript
import { useAccount } from 'wagmi';

export default function StatusCheck() {
  const { 
    address,        // User's wallet address
    isConnected,    // Connection status
    isConnecting,   // Connecting in progress
    isDisconnected  // Not connected
  } = useAccount();

  if (isConnecting) return <div>Connecting...</div>;
  if (isDisconnected) return <div>Please connect wallet</div>;
  if (isConnected) return <div>Connected as {address}</div>;
}
```

## Conditional Rendering Based on Connection

Show different UI based on wallet status:

```typescript
"use client";

import { useState, useEffect } from 'react';
import { useAccount } from 'wagmi';
import { Wallet, ConnectWallet } from '@coinbase/onchainkit/wallet';

export default function ConditionalUI() {
  const { isConnected } = useAccount();
  const [mounted, setMounted] = useState(false);

  // Prevent hydration mismatch
  useEffect(() => {
    setMounted(true);
  }, []);

  if (!mounted) {
    return (
      <div className="animate-pulse bg-amber-200/50 h-12 w-full rounded-xl" />
    );
  }

  return (
    <div>
      {isConnected ? (
        <div>
          <h2>Welcome!</h2>
          <button onClick={handlePurchase}>
            Purchase Policy
          </button>
        </div>
      ) : (
        <div className="p-6 bg-amber-50 border border-amber-100 rounded-2xl text-center">
          <p className="text-amber-900/80 text-sm mb-4">
            Connect your wallet to get started
          </p>
          <Wallet>
            <ConnectWallet className="w-full bg-amber-800 text-amber-50 py-3 rounded-xl justify-center font-medium">
              Connect Wallet
            </ConnectWallet>
          </Wallet>
        </div>
      )}
    </div>
  );
}
```

<Warning>
  Always use the `mounted` pattern to prevent hydration mismatches in Next.js. The server doesn't know wallet connection status, so initial render should match server output.
</Warning>

## Network Configuration

Ensure users are on the correct network:

```typescript
import { useAccount, useSwitchChain } from 'wagmi';
import { baseSepolia } from 'wagmi/chains';

export function NetworkCheck() {
  const { chain } = useAccount();
  const { switchChain } = useSwitchChain();

  if (chain?.id !== baseSepolia.id) {
    return (
      <div className="bg-red-50 border border-red-200 rounded-lg p-4">
        <p className="text-red-800 font-semibold">Wrong Network</p>
        <p className="text-red-700 text-sm mt-1">
          Please switch to Base Sepolia
        </p>
        <button
          onClick={() => switchChain({ chainId: baseSepolia.id })}
          className="mt-3 bg-red-600 text-white px-4 py-2 rounded"
        >
          Switch to Base Sepolia
        </button>
      </div>
    );
  }

  return null; // Correct network
}
```

### Sponsoring Gas (Paymaster)

Allow users to transact without holding ETH:

```typescript
import { useWriteContract } from 'wagmi';
import { ARIDGUARD_ADDRESS, ARIDGUARD_ABI } from '../utils/constants';
import { parseEther } from 'viem';

export function SponsoredPurchase() {
  const { writeContract } = useWriteContract();

  const handlePurchase = () => {
    writeContract({
      address: ARIDGUARD_ADDRESS,
      abi: ARIDGUARD_ABI,
      functionName: 'purchasePolicy',
      args: [5000, 360000],
      value: parseEther("0.0001"),
      // Paymaster configuration
      paymaster: {
        type: 'paymaster',
        url: process.env.NEXT_PUBLIC_PAYMASTER_URL,
      },
    });
  };

  return <button onClick={handlePurchase}>Purchase (Gas Free!)</button>;
}
```

<Note>
  Gas sponsorship requires a Paymaster service. Coinbase provides paymaster infrastructure for Base applications.
</Note>

## Error Handling

Handle common wallet errors:

```typescript
import { useAccount, useConnect } from 'wagmi';
import { useState } from 'react';

export function RobustWalletConnect() {
  const { isConnected } = useAccount();
  const { connect, connectors, error, isPending } = useConnect();
  const [customError, setCustomError] = useState<string>('');

  const handleConnect = async () => {
    try {
      setCustomError('');
      const connector = connectors[0]; // Coinbase Wallet
      connect({ connector });
    } catch (err) {
      setCustomError('Failed to connect wallet. Please try again.');
      console.error(err);
    }
  };

  return (
    <div>
      {error && (
        <div className="bg-red-50 text-red-800 p-3 rounded mb-4">
          {error.message}
        </div>
      )}
      {customError && (
        <div className="bg-red-50 text-red-800 p-3 rounded mb-4">
          {customError}
        </div>
      )}
      <button 
        onClick={handleConnect}
        disabled={isPending}
      >
        {isPending ? 'Connecting...' : 'Connect Wallet'}
      </button>
    </div>
  );
}
```
