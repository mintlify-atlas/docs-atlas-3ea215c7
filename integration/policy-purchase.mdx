## Overview

Purchasing a policy involves calling the `purchasePolicy` function on the AridGuard smart contract. This creates an on-chain insurance policy that the CRE workflow will monitor for drought conditions.

<Steps>
  <Step title="Connect Wallet">
    User connects their wallet via OnchainKit
  </Step>
  <Step title="Configure Parameters">
    Set latitude, longitude, and premium amount
  </Step>
  <Step title="Submit Transaction">
    Call `purchasePolicy` with coordinate parameters and ETH value
  </Step>
  <Step title="Wait for Confirmation">
    Transaction confirms on Base Sepolia (~2-5 seconds)
  </Step>
  <Step title="Policy Active">
    Policy is recorded on-chain and monitoring begins
  </Step>
</Steps>

## Smart Contract Function

The `purchasePolicy` function accepts GPS coordinates and a premium payment:

```solidity
function purchasePolicy(int256 lat, int256 long) external payable {
    require(msg.value > 0, "Premium must be greater than 0");
    
    bytes32 policyId = keccak256(
        abi.encodePacked(msg.sender, lat, long, block.timestamp)
    );
    
    policies[policyId] = Policy({
        farmer: msg.sender,
        lat: lat,
        long: long,
        premiumPaid: msg.value,
        isActive: true
    });
    
    activePolicyIds.push(policyId);
    
    emit PolicyPurchased(policyId, msg.sender, lat, long, msg.value);
}
```

### Parameters

- **`lat`** (int256): Latitude scaled by 100 (e.g., 0.05° = 5000)
- **`long`** (int256): Longitude scaled by 100 (e.g., 36.0° = 360000)
- **`msg.value`**: ETH premium amount (payable)

<Note>
  Coordinates are scaled by 100 to avoid floating-point numbers in Solidity. Latitude 0.05° is stored as 5000, and longitude 36.0° is stored as 360000.
</Note>

### Return Value

The function doesn't return a value but emits a `PolicyPurchased` event:

```solidity
event PolicyPurchased(
    bytes32 indexed policyId,
    address indexed farmer,
    int256 lat,
    int256 long,
    uint256 premium
);
```

## Basic Purchase Implementation

Use Wagmi's `useWriteContract` hook:

```typescript
"use client";

import { useWriteContract, useWaitForTransactionReceipt } from "wagmi";
import { parseEther } from "viem";
import { ARIDGUARD_ADDRESS, ARIDGUARD_ABI } from "../utils/constants";

export function PolicyPurchase() {
  const { data: hash, isPending, writeContract } = useWriteContract();
  
  const { isLoading: isConfirming, isSuccess } = useWaitForTransactionReceipt({
    hash,
  });

  const handlePurchase = () => {
    writeContract({
      address: ARIDGUARD_ADDRESS,
      abi: ARIDGUARD_ABI,
      functionName: 'purchasePolicy',
      args: [
        5000,    // Latitude: 0.05°
        360000   // Longitude: 36.0°
      ],
      value: parseEther("0.0001") // 0.0001 ETH premium
    });
  };

  return (
    <div>
      <button 
        onClick={handlePurchase}
        disabled={isPending || isConfirming}
      >
        {isPending ? "Confirm in Wallet..." : 
         isConfirming ? "Confirming..." : 
         "Purchase Policy"}
      </button>
      
      {isSuccess && (
        <div className="success-message">
          Policy purchased successfully!
        </div>
      )}
    </div>
  );
}
```

## Complete AridGuard Implementation

Here's the full implementation from the AridGuard frontend:

```typescript
"use client";

import { useState, useEffect } from "react";
import { useAccount, useWriteContract, useWaitForTransactionReceipt } from "wagmi";
import { parseEther } from "viem";
import { ArrowRight, ShieldCheck, ExternalLink } from "lucide-react";
import { ARIDGUARD_ADDRESS, ARIDGUARD_ABI } from "../utils/constants";

export default function PolicyPurchaseCard() {
  const { isConnected } = useAccount();
  const [purchased, setPurchased] = useState(false);
  const [mounted, setMounted] = useState(false);

  // Prevent hydration mismatch
  useEffect(() => {
    setMounted(true);
  }, []);

  // Contract interaction hooks
  const { data: hash, isPending: isWritePending, writeContract } = useWriteContract();
  const { isLoading: isConfirming, isSuccess: isConfirmed } = useWaitForTransactionReceipt({
    hash,
  });

  // Purchase handler
  const handlePurchase = () => {
    writeContract({
      address: ARIDGUARD_ADDRESS,
      abi: ARIDGUARD_ABI,
      functionName: 'purchasePolicy',
      args: [5000, 360000], // Kenya, Rift Valley
      value: parseEther("0.0001"),
    });
  };

  // Update UI when transaction confirms
  useEffect(() => {
    if (isConfirmed) {
      setPurchased(true);
    }
  }, [isConfirmed]);

  const loading = isWritePending || isConfirming;

  // Loading state during SSR
  if (!mounted) {
    return (
      <div className="p-6 bg-amber-50 border border-amber-100 rounded-2xl text-center animate-pulse">
        <div className="h-12 bg-amber-200/50 rounded-xl w-full"></div>
      </div>
    );
  }

  // Not connected state
  if (!isConnected) {
    return (
      <div className="p-6 bg-amber-50 border border-amber-100 rounded-2xl text-center">
        <p className="text-amber-900/80 text-sm mb-4">
          Connect your wallet to purchase a policy
        </p>
      </div>
    );
  }

  // Policy already purchased
  if (purchased) {
    return (
      <div className="bg-green-50 text-green-800 p-6 rounded-2xl border border-green-200 flex flex-col items-center justify-center text-center gap-3">
        <ShieldCheck size={32} className="text-green-600" />
        <div>
          <h4 className="font-bold">Policy Active</h4>
          <p className="text-sm text-green-700/80 mt-1">
            CRE workflow is monitoring rainfall for your coordinates.
          </p>
          {hash && (
            <a 
              href={`https://sepolia.basescan.org/tx/${hash}`} 
              target="_blank" 
              rel="noreferrer"
              className="mt-3 inline-flex items-center gap-1 text-xs font-semibold text-green-700 hover:text-green-900 underline"
            >
              View on BaseScan <ExternalLink size={12} />
            </a>
          )}
        </div>
      </div>
    );
  }

  // Purchase button
  return (
    <div className="space-y-3">
      <button 
        onClick={handlePurchase}
        disabled={loading}
        className="w-full py-4 bg-amber-800 hover:bg-amber-900 text-amber-50 font-semibold rounded-2xl transition-all flex items-center justify-center gap-2 disabled:opacity-70 shadow-lg shadow-amber-900/20"
      >
        {isWritePending ? "Confirm in Wallet..." : 
         isConfirming ? "Confirming on Base..." : 
         "Purchase Policy"}
        {!loading && <ArrowRight size={18} />}
      </button>
      
      {hash && !isConfirmed && (
        <p className="text-center text-xs text-amber-900/60">
          Transaction submitted. Waiting for confirmation...
        </p>
      )}
    </div>
  );
}
```

## Dynamic Coordinates

For production, allow users to select their farm location:

```typescript
import { useState } from 'react';
import { useWriteContract } from 'wagmi';
import { parseEther } from 'viem';

export function DynamicPolicyPurchase() {
  const [location, setLocation] = useState({ lat: 0, lng: 0 });
  const { writeContract } = useWriteContract();

  const handleLocationSelect = (lat: number, lng: number) => {
    setLocation({ lat, lng });
  };

  const handlePurchase = () => {
    writeContract({
      address: ARIDGUARD_ADDRESS,
      abi: ARIDGUARD_ABI,
      functionName: 'purchasePolicy',
      args: [
        Math.round(location.lat * 100),  // Scale by 100
        Math.round(location.lng * 100)
      ],
      value: parseEther("0.0001")
    });
  };

  return (
    <div>
      <MapPicker onLocationSelect={handleLocationSelect} />
      <div className="mt-4">
        <p>Selected: {location.lat}°, {location.lng}°</p>
        <button onClick={handlePurchase}>Purchase Policy</button>
      </div>
    </div>
  );
}
```

### Map Integration Example

```typescript
import { GoogleMap, Marker, useLoadScript } from '@react-google-maps/api';

interface MapPickerProps {
  onLocationSelect: (lat: number, lng: number) => void;
}

export function MapPicker({ onLocationSelect }: MapPickerProps) {
  const { isLoaded } = useLoadScript({
    googleMapsApiKey: process.env.NEXT_PUBLIC_GOOGLE_MAPS_KEY || ''
  });

  const [marker, setMarker] = useState({ lat: 0.05, lng: 36.0 });

  const handleMapClick = (e: google.maps.MapMouseEvent) => {
    if (e.latLng) {
      const lat = e.latLng.lat();
      const lng = e.latLng.lng();
      setMarker({ lat, lng });
      onLocationSelect(lat, lng);
    }
  };

  if (!isLoaded) return <div>Loading map...</div>;

  return (
    <GoogleMap
      zoom={8}
      center={marker}
      mapContainerClassName="w-full h-96 rounded-lg"
      onClick={handleMapClick}
    >
      <Marker position={marker} />
    </GoogleMap>
  );
}
```

## Error Handling

Handle common transaction errors:

```typescript
import { useWriteContract } from 'wagmi';
import { useState } from 'react';

export function RobustPolicyPurchase() {
  const { writeContract, error, isError } = useWriteContract();
  const [customError, setCustomError] = useState<string>('');

  const handlePurchase = async () => {
    try {
      setCustomError('');
      
      writeContract({
        address: ARIDGUARD_ADDRESS,
        abi: ARIDGUARD_ABI,
        functionName: 'purchasePolicy',
        args: [5000, 360000],
        value: parseEther("0.0001")
      });
    } catch (err: any) {
      console.error('Purchase failed:', err);
      
      // Custom error messages
      if (err.message?.includes('insufficient funds')) {
        setCustomError('Insufficient balance. Please add more ETH.');
      } else if (err.message?.includes('user rejected')) {
        setCustomError('Transaction cancelled.');
      } else {
        setCustomError('Transaction failed. Please try again.');
      }
    }
  };

  return (
    <div>
      {(isError || customError) && (
        <div className="bg-red-50 border border-red-200 text-red-800 p-4 rounded-lg mb-4">
          <p className="font-semibold">Error</p>
          <p className="text-sm mt-1">
            {customError || error?.message}
          </p>
        </div>
      )}
      
      <button onClick={handlePurchase}>
        Purchase Policy
      </button>
    </div>
  );
}
```

### Common Error Scenarios

<Accordion title="Insufficient Funds">
  **Error**: `"insufficient funds for gas * price + value"`
  
  **Cause**: User doesn't have enough ETH to cover premium + gas fees
  
  **Solution**:
  ```typescript
  // Check balance before transaction
  import { useBalance } from 'wagmi';
  
  const { data: balance } = useBalance({ address: userAddress });
  const premium = parseEther("0.0001");
  const estimatedGas = parseEther("0.00001"); // ~0.00001 ETH
  
  if (balance && balance.value < premium + estimatedGas) {
    alert('Insufficient balance');
    return;
  }
  ```
</Accordion>

<Accordion title="User Rejected Transaction">
  **Error**: `"User rejected the request"`
  
  **Cause**: User clicked "Reject" in wallet
  
  **Solution**: Handle gracefully, allow retry
  ```typescript
  if (err.message?.includes('user rejected')) {
    // Don't show as error, user intentionally cancelled
    return;
  }
  ```
</Accordion>

<Accordion title="Wrong Network">
  **Error**: `"chain mismatch"`
  
  **Cause**: User is connected to wrong network
  
  **Solution**:
  ```typescript
  import { useSwitchChain } from 'wagmi';
  import { baseSepolia } from 'wagmi/chains';
  
  const { switchChain } = useSwitchChain();
  
  try {
    await switchChain({ chainId: baseSepolia.id });
    // Then proceed with transaction
  } catch (err) {
    alert('Please switch to Base Sepolia network');
  }
  ```
</Accordion>

## Testing

### Local Testing with Hardhat

```typescript
import { expect } from "chai";
import { ethers } from "hardhat";

describe("PolicyPurchase", function () {
  it("Should purchase a policy", async function () {
    const [owner, farmer] = await ethers.getSigners();
    
    const AridGuard = await ethers.getContractFactory("AridGuard");
    const aridGuard = await AridGuard.deploy(owner.address);
    
    const premium = ethers.parseEther("0.0001");
    const lat = 5000;
    const long = 360000;
    
    await expect(
      aridGuard.connect(farmer).purchasePolicy(lat, long, { value: premium })
    )
      .to.emit(aridGuard, "PolicyPurchased")
      .withArgs(
        expect.any(String), // policyId
        farmer.address,
        lat,
        long,
        premium
      );
    
    // Verify policy was created
    const policyId = await aridGuard.activePolicyIds(0);
    const policy = await aridGuard.policies(policyId);
    
    expect(policy.farmer).to.equal(farmer.address);
    expect(policy.lat).to.equal(lat);
    expect(policy.long).to.equal(long);
    expect(policy.premiumPaid).to.equal(premium);
    expect(policy.isActive).to.be.true;
  });
});
```
